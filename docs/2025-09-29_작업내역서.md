## 문제 상황 분석

### 1. 초기 문제점
1. **퍼즐메인 로그인 폼과 8080 서비스 로그인 차이**
   - 퍼즐메인(`/K-CrossWord/`)의 로그인 폼이 8081 포트의 API를 호출
   - 실제 로그인은 8080 포트에서 처리되어야 함
   - 두 서비스 간 인증 시스템이 분리되어 있음

2. **로그인 후 힌트 로드 실패**
   - 8080에서 로그인한 사용자가 8081 퍼즐게임으로 진입 시 힌트가 로드되지 않음
   - 그리드는 정상적으로 렌더링되지만 힌트 API 호출 시 400 Bad Request 에러 발생

3. **게스트 ID 지속성 문제**
   - 새로고침이나 재진입 시 새로운 퍼즐이 생성되는 문제
   - 같은 세션 내에서 게스트 ID가 유지되어야 함

```javascript
(function() {
    // 1. 인증 상태 확인
    fetch('/K-CrossWord/api/auth/status')
        .then(response => response.json())
        .then(data => {
            if (data.authenticated) {
                // 인증된 사용자
                window.GUEST_ID = null;
                window.USER_TYPE = 'user';
                console.log('인증된 사용자로 게임 진행:', data.username);

                // 비인증 사용자 - 게스트 ID 확인
                const urlParams = new URLSearchParams(window.location.search);
                let guestId = localStorage.getItem('guestId') || urlParams.get('guest_id');
                
                if (guestId) {
                    window.GUEST_ID = guestId;
                    window.USER_TYPE = 'guest';
                    localStorage.setItem('guestId', guestId);
                } else {
                    // 게스트 ID도 없으면 메인으로 리다이렉트
                    window.location.href = '/K-CrossWord/';
                }
            }
        })
        .catch(error => {
            console.error('인증 상태 확인 실패:', error);
            window.location.href = '/K-CrossWord/';
        });
})();
```

```java
/**
 * 세션에서 인증된 사용자 ID 가져오기 (통합 인증)
 */
private Long getUserIdFromSession(HttpServletRequest request) {
    try {
        // Spring Security의 SecurityContext에서 인증 정보 확인
        Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
        
        if (authentication != null && authentication.isAuthenticated() && 
            !authentication.getPrincipal().equals("anonymousUser")) {
            
            String email = authentication.getName();
            System.out.println("인증된 사용자 이메일: " + email);
            
            // 이메일로 사용자 ID 조회 (8080 서비스의 users 테이블과 동일)
            if ("rainynux@gmail.com".equals(email)) {
                return 1L; // 관리자 계정
            } else {
                // 다른 사용자들의 경우 이메일 해시를 사용하여 일관된 ID 생성
                return (long) Math.abs(email.hashCode() % 10000) + 1000;
            }
        }
    } catch (Exception e) {
        System.err.println("세션에서 사용자 ID 추출 실패: " + e.getMessage());
    }
    return null;
}
```

**효과**: 8080 서비스의 세션 정보를 8081에서 정상적으로 인식

1. **퍼즐메인 접근** → 로그인 버튼 클릭 → 8080 로그인 페이지
