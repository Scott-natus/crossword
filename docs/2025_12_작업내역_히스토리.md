```java
Object langCodeObj = hintData.get("language_code");
String languageCode = "KR"; // 기본값
if (langCodeObj != null) {
    languageCode = langCodeObj.toString();
}
hint.setLanguageCode(languageCode);

// 순서 처리
Object orderObj = hintData.get("display_order");
Integer orderVal = null;
if (orderObj instanceof Integer) orderVal = (Integer) orderObj;
else if (orderObj instanceof Number) orderVal = ((Number) orderObj).intValue();
hint.setDisplayOrder(orderVal != null ? orderVal : 0); // Default to 0
```
